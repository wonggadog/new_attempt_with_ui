### âœ… JavaScript Snippet for Dummy DB + Table Toggle

Add this **at the end of `scripts.js`**:

```javascript
document.addEventListener("DOMContentLoaded", function () {
  // Simulated "database" data
  const dummyHistory = []; // Leave empty to simulate no history

  const tableSection = document.querySelector(".table-responsive");
  const historyPlaceholder = document.getElementById("history-placeholder");

  if (dummyHistory.length === 0) {
    // No data â€“ show placeholder
    tableSection.style.display = "none";
    historyPlaceholder.style.display = "block";
  } else {
    // Data present â€“ show table
    tableSection.style.display = "block";
    historyPlaceholder.style.display = "none";

    const tableBody = document.querySelector("table tbody");
    dummyHistory.forEach(entry => {
      const row = document.createElement("tr");
      row.classList.add("clickable-row");
      row.setAttribute("data-href", entry.link);
      row.innerHTML = `
        <td>
          <div class="d-flex align-items-center">
            <img src="${entry.profile}" alt="Profile" class="rounded-circle me-2" width="40" height="40">
            <div>
              <div class="fw-semibold">${entry.name}</div>
              <small class="text-muted">${entry.position}</small>
            </div>
          </div>
        </td>
        <td><span class="badge bg-${entry.statusColor} text-dark">${entry.status}</span></td>
        <td>${entry.date}</td>
        <td>${entry.time}</td>
        <td>${entry.type}</td>
      `;
      tableBody.appendChild(row);
    });
  }
});
```
